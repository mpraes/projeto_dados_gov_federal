CREATE TABLE TA_ITEM_COMPRA (
   ID_ORGAO INT NOT NULL ,
   ID_UG INT NOT NULL ,
   ID_ITEM_COMPRA INT NOT NULL,
   ID_CONTRATO INT NOT NULL,
   QUANTIDADE_ITEM DECIMAL(12,2) ,
   VALOR_ITEM DECIMAL(12,2),
   FOREIGN KEY (ID_ORGAO) REFERENCES DIM_ORGAO (ID_ORGAO),
   FOREIGN KEY (ID_UG) REFERENCES DIM_UNIDADE_GESTORA (ID_UG),
   FOREIGN KEY (ID_ITEM_COMPRA) REFERENCES DIM_ITEM_COMPRA (ID_ITEM_COMPRA),
   FOREIGN KEY (ID_CONTRATO) REFERENCES DIM_CONTRATO(ID_CONTRATO)
);



CREATE TABLE TA_TERMO_ADITIVO (
   ID_CONTRATO INT NOT NULL,
   ID_UG INT NOT NULL,
   ID_ORGAO_SUPERIOR INT NOT NULL,
   ID_ORGAO INT NOT NULL,
   ID_ADITIVO INT NOT NULL,
   DATA_PUBLICACAO DATE,
   FOREIGN KEY (ID_CONTRATO) REFERENCES DIM_CONTRATO (ID_CONTRATO),
   FOREIGN KEY (ID_UG) REFERENCES DIM_UNIDADE_GESTORA (ID_UG),
   FOREIGN KEY (ID_ORGAO_SUPERIOR) REFERENCES DIM_ORGAO_SUPERIOR (ID_ORGAO_SUPERIOR),
   FOREIGN KEY (ID_ORGAO) REFERENCES DIM_ORGAO (ID_ORGAO),
   FOREIGN KEY (ID_ADITIVO) REFERENCES DIM_ADITIVO (ID_ADITIVO)
);
