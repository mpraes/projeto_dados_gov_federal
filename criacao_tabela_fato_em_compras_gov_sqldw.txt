CREATE TABLE FCT_COMPRAS (
   ID_CONTRATO INT NOT NULL,
   ID_FUNDAMENTO INT NOT NULL,
   ID_MODALIDADE INT NOT NULL,
   ID_SITUACAO INT NOT NULL,
   ID_ORGAO_SUPERIOR INT NOT NULL,
   ID_CNPJ_CONTRATADO INT NOT NULL,
   ID_UG_LICITACAO INT NOT NULL,
   ID_MODALIDADE_SITUACAO INT NOT NULL,
   ID_ORGAO INT NOT NULL,
   ID_UG INT NOT NULL,
   DATA_ASSINATURA_CONTRATO DATE,
   DATA_PUBLICACAO_DOU DATE,
   DATA_INICIO_VIGENCIA DATE,
   DATA_FIM_VIGENCIA DATE,
   VALOR_INICIAL_COMPRA DECIMAL(12,2),
   VALOR_FINAL_COMPRA DECIMAL(12,2),
   NUMERO_LICITACAO INT,
   FOREIGN KEY (ID_CONTRATO) REFERENCES DIM_CONTRATO (ID_CONTRATO),
   FOREIGN KEY (ID_FUNDAMENTO) REFERENCES DIM_FUNDAMENTO_LEGAL (ID_FUNDAMENTO),
   FOREIGN KEY (ID_MODALIDADE) REFERENCES DIM_MOD_COMPRA (ID_MODALIDADE),
   FOREIGN KEY (ID_SITUACAO) REFERENCES DIM_SITUACAO_CONTRATO (ID_SITUACAO),
   FOREIGN KEY (ID_ORGAO_SUPERIOR) REFERENCES DIM_ORGAO_SUPERIOR (ID_ORGAO_SUPERIOR),
   FOREIGN KEY (ID_CNPJ_CONTRATADO) REFERENCES DIM_SITUACAO_CONTRATO (ID_SITUACAO),
   FOREIGN KEY (ID_UG_LICITACAO) REFERENCES DIM_UG_LICITACAO (ID_UG_LICITACAO),
   FOREIGN KEY (ID_MODALIDADE_SITUACAO) REFERENCES DIM_MODALIDADE_LICITACAO (ID_MOD_LICITACAO),
   FOREIGN KEY (ID_ORGAO) REFERENCES DIM_ORGAO (ID_ORGAO),
   FOREIGN KEY (ID_UG) REFERENCES DIM_UNIDADE_GESTORA (ID_UG)
);
